public class Solution {
    public int maxProfit(int[] prices) {
        
        if (prices.length < 2) return 0;
        
        int maxprofit = 0;
        int curmin = prices[0];
        for (int i = 1; i < prices.length; i++) {
            curmin = Math.min(curmin,prices[i]);
            maxprofit = Math.max(maxprofit, prices[i]-curmin);
        }
        return maxprofit;
    }
}
